620  int chunkHeight,
621  bool *ok
622 ) {
623  int width;
624  int height;
625  QImage image(imagePath);
626  QFileInfo rootImage(imagePath);
627  QList<QImage> imageChunks;
628  QStringList chunks;
629  QString fileName;
630 
631  // Create chunks of image.
632  for (int xOffset = 0; xOffset < image.width(); xOffset += chunkWidth) {
633  width = xOffset + chunkWidth > image.width() ? image.width() - xOffset : chunkWidth;
634 
635  for (int yOffset = 0; yOffset <= image.height(); yOffset += chunkHeight) {
636  height = yOffset + chunkHeight > image.height() ? image.height() - yOffset : chunkHeight;
637 
638  imageChunks << image.copy(xOffset, yOffset, width, height);
639  }
640  }
641 
642  // Save images on disk.
643  for (int i = 0; i < imageChunks.size(); ++i) {
644  fileName = Utils::getTmpDirectory() + "/" + rootImage.baseName() + "_chunk_" + QString::number(i+1) + '.' + rootImage.suffix();
645 
646  // Only returns chunks saved succesfully.
647  if (imageChunks.at(i).save(fileName, Q_NULLPTR, 100)) {
648  chunks << fileName;
649  Log::writeLog(QString(Q_FUNC_INFO) + " Saved image chunk: " + fileName);
650  } else {
651  Log::writeError(QString(Q_FUNC_INFO) + " Could no saved image chunk: " + fileName);
652  if (ok) {
653  *ok = false;
654  }
655  }
656  }
