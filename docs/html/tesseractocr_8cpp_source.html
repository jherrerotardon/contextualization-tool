186 QStringList TesseractOcr::processExtration(const QString &source)
187 {
188  QStringList result;
189  QString text = source; // Because future replace will modify the string.
190 
191  if (source.isEmpty()) {
192  return result;
193  }
194 
195  // Remove rare characters (Slug)
196  text = text.replace(QRegularExpression("[^a-zA-Z0-9 -,\.]"), "\n");
197 
198  // Force the beginning of the string to be a one-word character.
199  text = text.replace(QRegularExpression("^\\W+\\s*", QRegularExpression::MultilineOption), "");
200 
201  // If the first letter of a sentence is lowercase, it is assumed that it belongs to the previous word.
202  text = text.replace(QRegularExpression("\n([a-z])"), " \\1");
203 
204  result << text.split('\n',QString::SkipEmptyParts);
205 
206  result.removeDuplicates();
207 
208  // Returns only those strings that have at least any character of a word.
209  return result.filter(QRegularExpression("\\w+"));
210 }
